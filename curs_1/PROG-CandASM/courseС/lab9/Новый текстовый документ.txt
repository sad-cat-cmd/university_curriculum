	#define MAX_SIZE 10
	typedef struct {
	int N;
	char name[10];
	struct{
		int minSize;
		int maxSize;
	}re;
	int age;
	}Person;

	void fillTable(Person table[], int size) {
	for (int i = 0; i<size; i++) {
	printf("Введите данные для структуры %d:\n", i + 1);
	printf("Название: ");
	scanf_s("%s", &table[i].name);
	printf("Минимальныйразмер: ");
	scanf_s("%d", &table[i].re.minSize);
	printf("Максимальныйразмер: ");
	scanf_s("%d", &table[i].re.maxSize);
	printf("Цена: ");
	scanf_s("%d", &table[i].age);
		}
	}

	void printTable(Person table[], int size) {
	const char head[] =
	"+-------------------------------------------------------------+\n"
	"|   |                 |       размер       |                  |\n"
	"| # |    название     |----------+---------|       цена       |\n"
	"|   |                 |    мин|  макс   |                  |\n"
	"+---+-----------------+----------+---------+------------------+\n";
	printf("%s", head);
	for (int i = 0; i< size; i++) {
	printf("| %1d | %-15s |   %-4d   |   %-4d  | %-15d  |\n", i+1, table[i].name, table[i].re.minSize, table[i].re.maxSize, table[i].age);
			printf("+---+-----------------+----------+---------+------------------+\n");
		}
	}

	int main() {
	setlocale(LC_ALL, "Russian");
	setlocale(LC_ALL, "rus");
	Person table[MAX_SIZE];
	int size;
	printf("Введите количество строк таблицы: ");
	scanf("%d", &size);

	if (size >MAX_SIZE) {
	printf("Ошибка! Превышен максимальный размер таблицы.\n");
	return 0;
		}

	fillTable(table, size);
	printTable(table, size);
